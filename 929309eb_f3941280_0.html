AT(.com_text*)
const char strf[] = "[F_0x%X]";
AT(.com_text*)
const char strr[] = "[R_0x%X]";

AT(.com_text.port_int)
void port_int_isr(void)
{
    if (WKUPEDG & BIT(22)) {  //fall
        printf(strf,WKUPEDG);
        WKUPCPND = BIT(22);
    }

    if (WKUPEDG & BIT(23)) { //rise
        printf(strr,WKUPEDG);
        WKUPCPND = BIT(23);
    }

void port_int_init(void)
{
    printf("port_int_init\n");
    GPIOFDE  |= BIT(5);
    GPIOFDIR |= BIT(5);
    GPIOFPU  |= BIT(5);

    PORTINTEN |= BIT(26);
    WKUPCPND = 0xff << 16;

//    //上升级沿产生中断示例
    PORTINTEDG &= ~BIT(26); //rise   
    WKUPEDG &= ~BIT(6);     //rise
    WKUPCON |= BIT(7);     //int_rise
    WKUPCON |= BIT(16);    //INT EN

//    //下降级沿产生中断示例

//    WKUPCON |= BIT(6)|BIT(7); //int rise or fall
//    WKUPCON |= BIT(16);       //INT EN

    register_isr(IRQ_PORT_INT_VECTOR,port_int_isr);
    PICPR &= ~BIT(IRQ_PORT_INT_VECTOR);
    PICEN |= BIT(IRQ_PORT_INT_VECTOR);
    WDT_DIS(); while(1);
